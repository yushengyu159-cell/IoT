## 自动化测试脚本

### Bash 测试脚本

```bash
#!/bin/bash

# 文件上传接口测试脚本
# 测试账号: 18027473816@163.com
# 接口地址: https://app.esgvisa.com/api/esg/upload-encrypted

API_URL="https://app.esgvisa.com/api/esg/upload-encrypted"
UPLOADER="18027473816@163.com"
TEST_FILE="test.pdf"

echo "=========================================="
echo "文件上传接口测试"
echo "接口地址: $API_URL"
echo "测试账号: $UPLOADER"
echo "测试文件: $TEST_FILE"
echo "=========================================="
echo ""

# 检查测试文件是否存在
if [ ! -f "$TEST_FILE" ]; then
    echo "❌ 错误: 测试文件 $TEST_FILE 不存在"
    echo "请先创建一个测试文件"
    exit 1
fi

echo "📤 开始上传文件..."
echo ""

# 执行上传
RESPONSE=$(curl -X POST "$API_URL" \
  -F "file=@$TEST_FILE" \
  -F "desc=自动化测试文件上传" \
  -F "uploader=$UPLOADER" \
  -w "\nHTTP_CODE:%{http_code}" \
  -s)

# 提取HTTP状态码
HTTP_CODE=$(echo "$RESPONSE" | grep -o "HTTP_CODE:[0-9]*" | cut -d: -f2)
BODY=$(echo "$RESPONSE" | sed 's/HTTP_CODE:[0-9]*$//')

echo "HTTP状态码: $HTTP_CODE"
echo ""
echo "响应内容:"
echo "$BODY" | jq '.' 2>/dev/null || echo "$BODY"
echo ""

# 检查上传结果
if [ "$HTTP_CODE" = "200" ]; then
    CODE=$(echo "$BODY" | jq -r '.code' 2>/dev/null)
    if [ "$CODE" = "200" ]; then
        echo "✅ 上传成功!"
        
        # 提取关键信息
        CID=$(echo "$BODY" | jq -r '.data.meta.cids[0]' 2>/dev/null)
        KEY=$(echo "$BODY" | jq -r '.data.key' 2>/dev/null)
        FILE_NAME=$(echo "$BODY" | jq -r '.data.meta.fileName' 2>/dev/null)
        FILE_SIZE=$(echo "$BODY" | jq -r '.data.meta.fileSize' 2>/dev/null)
        CHUNK_COUNT=$(echo "$BODY" | jq -r '.data.chunkCount' 2>/dev/null)
        TOTAL_TIME=$(echo "$BODY" | jq -r '.data.timeStats.totalTime' 2>/dev/null)
        
        echo ""
        echo "📋 文件信息:"
        echo "  文件名: $FILE_NAME"
        echo "  文件大小: $FILE_SIZE 字节"
        echo "  分片数量: $CHUNK_COUNT"
        echo "  上传耗时: $TOTAL_TIME"
        echo ""
        echo "🔑 关键数据:"
        echo "  CID: $CID"
        echo "  加密密钥: $KEY"
        echo ""
        echo "⚠️  请妥善保管加密密钥，用于后续文件下载"
        
        # 保存结果到文件
        echo "$BODY" > "upload_result_$(date +%Y%m%d_%H%M%S).json"
        echo ""
        echo "✅ 响应已保存到文件"
    else
        MESSAGE=$(echo "$BODY" | jq -r '.message' 2>/dev/null)
        echo "❌ 上传失败: $MESSAGE"
        exit 1
    fi
else
    echo "❌ HTTP错误: $HTTP_CODE"
    exit 1
fi

echo ""
echo "=========================================="
echo "测试完成"
echo "=========================================="
```

### Python 测试脚本

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-

"""
文件上传接口测试脚本
测试账号: 18027473816@163.com
接口地址: https://app.esgvisa.com/api/esg/upload-encrypted
"""

import requests
import json
import sys
import os
from datetime import datetime

# 配置信息
API_URL = "https://app.esgvisa.com/api/esg/upload-encrypted"
UPLOADER = "18027473816@163.com"
TEST_FILE = "test.pdf"

def upload_file(file_path, desc="自动化测试文件上传", uploader=UPLOADER):
    """
    上传文件到服务器
    
    Args:
        file_path: 文件路径
        desc: 文件描述
        uploader: 上传者账号
    
    Returns:
        dict: 响应数据
    """
    print("=" * 50)
    print("文件上传接口测试")
    print("=" * 50)
    print(f"接口地址: {API_URL}")
    print(f"测试账号: {uploader}")
    print(f"测试文件: {file_path}")
    print("=" * 50)
    print()
    
    # 检查文件是否存在
    if not os.path.exists(file_path):
        print(f"❌ 错误: 文件 {file_path} 不存在")
        return None
    
    # 获取文件大小
    file_size = os.path.getsize(file_path)
    print(f"📄 文件信息:")
    print(f"  文件名: {os.path.basename(file_path)}")
    print(f"  文件大小: {file_size:,} 字节 ({file_size / 1024 / 1024:.2f} MB)")
    print()
    
    # 准备上传数据
    files = {
        'file': (os.path.basename(file_path), open(file_path, 'rb'), 'application/octet-stream')
    }
    data = {
        'desc': desc,
        'uploader': uploader
    }
    
    print("📤 开始上传文件...")
    print()
    
    try:
        # 发送请求
        response = requests.post(
            API_URL,
            files=files,
            data=data,
            timeout=300  # 5分钟超时
        )
        
        # 关闭文件
        files['file'][1].close()
        
        # 检查HTTP状态码
        print(f"HTTP状态码: {response.status_code}")
        print()
        
        # 解析响应
        try:
            result = response.json()
        except json.JSONDecodeError:
            print("❌ 错误: 响应不是有效的JSON格式")
            print(f"响应内容: {response.text[:500]}")
            return None
        
        print("响应内容:")
        print(json.dumps(result, indent=2, ensure_ascii=False))
        print()
        
        # 检查业务状态码
        if response.status_code == 200 and result.get('code') == 200:
            print("✅ 上传成功!")
            print()
            
            # 提取关键信息
            data_info = result.get('data', {})
            meta = data_info.get('meta', {})
            
            print("📋 文件信息:")
            print(f"  文件名: {meta.get('fileName', 'N/A')}")
            print(f"  文件大小: {meta.get('fileSize', 0):,} 字节")
            print(f"  分片数量: {data_info.get('chunkCount', 0)}")
            print(f"  上传时间: {meta.get('uploadAt', 'N/A')}")
            
            time_stats = data_info.get('timeStats', {})
            if time_stats:
                print(f"  上传耗时: {time_stats.get('totalTime', 'N/A')}")
            
            print()
            print("🔑 关键数据:")
            cids = meta.get('cids', [])
            if cids:
                print(f"  主CID: {cids[0]}")
            print(f"  加密密钥: {data_info.get('key', 'N/A')}")
            print()
            print("⚠️  请妥善保管加密密钥，用于后续文件下载")
            
            # 保存结果到文件
            timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
            result_file = f"upload_result_{timestamp}.json"
            with open(result_file, 'w', encoding='utf-8') as f:
                json.dump(result, f, indent=2, ensure_ascii=False)
            print()
            print(f"✅ 响应已保存到文件: {result_file}")
            
            return result
        else:
            print("❌ 上传失败:")
            print(f"  错误码: {result.get('code', 'N/A')}")
            print(f"  错误消息: {result.get('message', 'N/A')}")
            return None
            
    except requests.exceptions.Timeout:
        print("❌ 错误: 请求超时（超过5分钟）")
        return None
    except requests.exceptions.RequestException as e:
        print(f"❌ 网络错误: {e}")
        return None
    except Exception as e:
        print(f"❌ 未知错误: {e}")
        import traceback
        traceback.print_exc()
        return None
    finally:
        print()
        print("=" * 50)
        print("测试完成")
        print("=" * 50)

def test_multiple_files():
    """测试上传多个文件"""
    test_files = [
        ("test.pdf", "PDF文档测试"),

        else:

if __name__ == "__main__":
    # 检查命令行参数
    if len(sys.argv) > 1:
        file_path = sys.argv[1]
        desc = sys.argv[2] if len(sys.argv) > 2 else "命令行测试上传"

    else:
        # 使用默认测试文件
        if os.path.exists(TEST_FILE):
            upload_file(TEST_FILE)
        else:
            print(f"❌ 错误: 默认测试文件 {TEST_FILE} 不存在")

            print("或创建测试文件: test.pdf")
```

### JavaScript/Node.js 测试脚本

```javascript
/**
 * 文件上传接口测试脚本
 * 测试账号: 18027473816@163.com
 * 接口地址: https://app.esgvisa.com/api/esg/upload-encrypted
 * 
 * 使用方法:
 * npm install form-data axios
 * node test_upload.js [文件路径]
 */

const fs = require('fs');
const path = require('path');
const FormData = require('form-data');
const axios = require('axios');

// 配置信息
const API_URL = 'https://app.esgvisa.com/api/esg/upload-encrypted';
const UPLOADER = '18027473816@163.com';
const DEFAULT_TEST_FILE = 'test.pdf';

/**
 * 上传文件
 * @param {string} filePath - 文件路径
 * @param {string} desc - 文件描述
 * @param {string} uploader - 上传者账号
 */
async function uploadFile(filePath, desc = '自动化测试文件上传', uploader = UPLOADER) {
    console.log('='.repeat(50));
    console.log('文件上传接口测试');
    console.log('='.repeat(50));
    console.log(`接口地址: ${API_URL}`);
    console.log(`测试账号: ${uploader}`);
    console.log(`测试文件: ${filePath}`);
    console.log('='.repeat(50));
    console.log();

    // 检查文件是否存在
    if (!fs.existsSync(filePath)) {
        console.error(`❌ 错误: 文件 ${filePath} 不存在`);
        return null;
    }

    // 获取文件信息
    const stats = fs.statSync(filePath);
    const fileName = path.basename(filePath);
    const fileSize = stats.size;

    console.log('📄 文件信息:');
    console.log(`  文件名: ${fileName}`);
    console.log(`  文件大小: ${fileSize.toLocaleString()} 字节 (${(fileSize / 1024 / 1024).toFixed(2)} MB)`);
    console.log();

    // 创建FormData
    const formData = new FormData();
    formData.append('file', fs.createReadStream(filePath), fileName);
    formData.append('desc', desc);
    formData.append('uploader', uploader);

    console.log('📤 开始上传文件...');
    console.log();

    try {
        // 发送请求
        const response = await axios.post(API_URL, formData, {
            headers: formData.getHeaders(),
            maxContentLength: Infinity,
            maxBodyLength: Infinity,
            timeout: 300000, // 5分钟超时
        });

        console.log(`HTTP状态码: ${response.status}`);
        console.log();
        console.log('响应内容:');
        console.log(JSON.stringify(response.data, null, 2));
        console.log();

        // 检查业务状态码
        if (response.status === 200 && response.data.code === 200) {
            console.log('✅ 上传成功!');
            console.log();

            const data = response.data.data;
            const meta = data.meta;

            console.log('📋 文件信息:');
            console.log(`  文件名: ${meta.fileName}`);
            console.log(`  文件大小: ${meta.fileSize.toLocaleString()} 字节`);
            console.log(`  分片数量: ${data.chunkCount}`);
            console.log(`  上传时间: ${meta.uploadAt}`);

            if (data.timeStats) {
                console.log(`  上传耗时: ${data.timeStats.totalTime}`);
            }

            console.log();
            console.log('🔑 关键数据:');
            if (meta.cids && meta.cids.length > 0) {
                console.log(`  主CID: ${meta.cids[0]}`);
            }
            console.log(`  加密密钥: ${data.key}`);
            console.log();
            console.log('⚠️  请妥善保管加密密钥，用于后续文件下载');

            // 保存结果到文件
            const timestamp = new Date().toISOString().replace(/[:.]/g, '-').slice(0, -5);
            const resultFile = `upload_result_${timestamp}.json`;
            fs.writeFileSync(resultFile, JSON.stringify(response.data, null, 2), 'utf8');
            console.log();
            console.log(`✅ 响应已保存到文件: ${resultFile}`);

            return response.data;
        } else {
            console.log('❌ 上传失败:');
            console.log(`  错误码: ${response.data.code || 'N/A'}`);
            console.log(`  错误消息: ${response.data.message || 'N/A'}`);
            return null;
        }
    } catch (error) {
        console.error('❌ 错误:');
        if (error.response) {
            // 服务器返回了错误响应
            console.error(`  HTTP状态码: ${error.response.status}`);
            console.error(`  错误信息: ${JSON.stringify(error.response.data, null, 2)}`);
        } else if (error.request) {
            // 请求已发送但没有收到响应
            console.error('  网络错误: 无法连接到服务器');
        } else {
            // 其他错误
            console.error(`  错误: ${error.message}`);
        }
        return null;
    } finally {
        console.log();
        console.log('='.repeat(50));
        console.log('测试完成');
        console.log('='.repeat(50));
    }
}

// 主函数
async function main() {
    const filePath = process.argv[2] || DEFAULT_TEST_FILE;
    const desc = process.argv[3] || '自动化测试文件上传';

    if (!fs.existsSync(filePath)) {
        console.error(`❌ 错误: 文件 ${filePath} 不存在`);
        console.log('使用方法: node test_upload.js [文件路径] [描述]');
        process.exit(1);
    }

    await uploadFile(filePath, desc);
}

// 运行测试
if (require.main === module) {
    main().catch(console.error);
}

module.exports = { uploadFile };
```
