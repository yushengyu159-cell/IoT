    ctx := context.Background()
    db := g.DB()
    
    // 查询所有已注册的用户邮箱（状态为 completed 或 verified）
    result, err := db.Model("dids").Fields("email").

    // 提取邮箱地址

        if email != "" {

        Total:  len(emails),
        Emails: emails,

}

    ctx := context.Background()
    
    if email == "" {
        return nil, fmt.Errorf("邮箱地址不能为空")

    // 从数据库获取用户详细信息
    db := g.DB()
